asm void
PPCEieio ()
{
#ifdef __MWERKS__
    nofralloc;
    mfmsr  r5;
    rlwinm r6, r5, 0, 17, 15;
    mtmsr  r6;
    mfspr  r3, HID0;
    ori    r4, r3, 0x8;
    mtspr  HID0, r4;
    isync;
    eieio;
    isync;
    mtspr HID0, r3;
    mtmsr r5;
    isync;
    blr;
#endif
}

asm void
PPCHalt ()
{
#ifdef __MWERKS__
    nofralloc;

    sync;

loop:
    nop;
    li r3, 0;
    nop;
    b loop;
#endif
}
