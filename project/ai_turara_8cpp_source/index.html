void
AITurara::doBehaviorInit ()
{
    AITurara_State* state = AITurara_Behaviors;
    AITurara_State* end = AITurara_Behaviors + 2;

    AITurara_State* t;

    while (true)
    {
        if (state->StateId == mCurState)
        {
            t = state;
            break;
        }

        state++;

        if (state == end)
        {
            t = NULL;
            break;
        }
    }

    if (t != NULL)
    {
        (this->*t->InitFunc)();
    }
}
