inline DSError TRKAppendBuffer1_ui8(TRKBuffer* buffer, const u8 data)
{
    if (buffer->position >= TRKMSGBUF_SIZE) {
        return DS_MessageBufferOverflow;
    }

    buffer->data[buffer->position++] = data;
    buffer->length++;
    return DS_NoError;
}
